# Cum se configureaza?
Script-ul se configureaza foarte usor avand un sistem de config la indemana pentru toata lumea.<br />
local Config = {<br />
    &nbspInteractSound = true, -- Default este pe true dar daca nu aveti sistemul de InteractSound atunci puneti-l pe false<br />
    &nbspSoundVolume = 0.8, -- Volumul muzici in timp ce colinzi<br />
    &nbspBagId = 45, -- Aici este id-ul genti de care aveti nevoie pentru a face quest-ul<br />
    &nbspTure = 5, -- Aici puneti de cate ori vreti sa faceti quest-ul<br />
    &nbspMarker = 0 -- Aici puneti ID-ul marker-ului pe care doriti sa il afisezi in momentul in care sunteti la o casa<br />
}<br />
